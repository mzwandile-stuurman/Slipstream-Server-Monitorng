CREATE TABLE process_monitor (
    id SERIAL PRIMARY KEY,
    name TEXT,
    session_id INT,
    ws BIGINT,
    vm BIGINT,
    private_mem BIGINT,
    working_set BIGINT,
    virtual_mem BIGINT,
    paged_mem BIGINT,
    peak_ws BIGINT,
    peak_vm BIGINT,
    max_ws BIGINT,
    cpu DOUBLE PRECISION,
    priv_proc_time DOUBLE PRECISION,
    user_proc_time DOUBLE PRECISION,
    total_proc_time DOUBLE PRECISION,
    processor_affinity TEXT,
    start_time TIMESTAMP,
    has_exited BOOLEAN,
    responding BOOLEAN,
    machine_name TEXT,
    read_bytes BIGINT,
    write_bytes BIGINT,
    other_bytes BIGINT,
    total_disk_io BIGINT,
    timestamp TIMESTAMP
);
